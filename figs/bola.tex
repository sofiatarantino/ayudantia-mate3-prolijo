\begin{center}
\begin{tikzpicture}
    % define the “δ” length once
    \def\deltalength{1.6cm}
  
    %—— Left: circle of radius δ —————————————————————————————
    \begin{scope}
      % filled circle
      \filldraw[fill=blue!20, draw=blue!20, thick]
        (0,0) circle [radius=\deltalength];
    
        % draw R^2 axes
        \draw[-Stealth, thin] (-1.5*\deltalength,0) -- (1.5*\deltalength,0) node[right] {$x$};
        \draw[-Stealth, thin] (0,-1.5*\deltalength) -- (0,1.5*\deltalength) node[above] {$y$};

        % ticks and labels at ±δ on x
        \draw (\deltalength, 0.05) -- ++(0,-0.1) node[below] {$\delta$};
        \draw (-\deltalength, 0.05) -- ++(0,-0.1) node[below] {$-\delta$};
        % ticks and labels at ±δ on y
        \draw (0.05, \deltalength) -- ++(-0.1,0) node[left] {$\delta$};
        \draw (0.05,-\deltalength) -- ++(-0.1,0) node[left] {$-\delta$};


        % label underneath
      \node[below,yshift=-2pt] at (0,-1.5*\deltalength){Bola con distancia inducida por la norma 2};
    
    \end{scope}
  
    %—— Right: rhombus of “radius” δ ————————————————————————
    \begin{scope}[xshift=7cm]
      % filled rhombus (diamond) with half-diagonals = δ
      \filldraw[fill=red!20, draw=red!20, thick] 
        (0,\deltalength) -- (\deltalength,0) -- (0,-\deltalength) -- (-\deltalength,0) -- cycle;

        % draw R^2 axes
      \draw[-Stealth, thin] (-1.5*\deltalength,0) -- (1.5*\deltalength,0) node[right] {$x$};
      \draw[-Stealth, thin] (0,-1.5*\deltalength) -- (0,1.5*\deltalength) node[above] {$y$};

        % ticks and labels at ±δ on x
        \draw (\deltalength, 0.05) -- ++(0,-0.1) node[below] {$\delta$};
        \draw (-\deltalength, 0.05) -- ++(0,-0.1) node[below] {$-\delta$};
        % ticks and labels at ±δ on y
        \draw (0.05, \deltalength) -- ++(-0.1,0) node[left] {$\delta$};
        \draw (0.05,-\deltalength) -- ++(-0.1,0) node[left] {$-\delta$};

      % label underneath
      \node[below,yshift=-2pt] at (0,-1.5*\deltalength){Bola con distancia inducida por la norma 1};
    \end{scope}
  \end{tikzpicture}
