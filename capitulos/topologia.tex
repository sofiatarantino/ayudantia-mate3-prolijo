\begin{definition}\label{def:norma-general}
    Una \emph{norma} en $\mathbb{R}^n$ es un campo escalar en $\mathbb{R}^n$, el cual notaremos $\| \; \|$,  que  satisface las siguientes condiciones:  dados $\mathbf{x},\;\mathbf{y} \in \mathbb{R}^n$ y $\lambda \in \mathbb{R}$:
    \begin{enumerate}
        \item Positividad: $\|\mathbf{x}\| \geq 0\;\land\;\|\mathbf{x}\| = 0 \iff \mathbf{x} = \boldsymbol{0}$.
        \item Homogeneidad: $\|\lambda \mathbf{x}\| = |\lambda|\|\mathbf{x}\|$.
        \item Desigualdad triangular: $\|\mathbf{x} + \mathbf{y}\| \leq \|\mathbf{x}\| + \|\mathbf{y}\|$.
    \end{enumerate}
\end{definition}

\begin{obs}
  En $\Rn{n}$ existen infinitas campos escalares que cumplen con las condiciones reci\'en mecionadas, mostraremos algunas de ellas a continuaci\'on.
\end{obs}

\begin{definition}\label{def:norm1} Se define la \emph{norma 1},  la cual notaremos por $\| \; \|_1$,  a 
    \[
        \|\mathbf{x}\|_1 = \sum_{i=1}^n |x_i|,
    \]
    donde $\mathbf{x}=(x_1,\ldots,x_n)$. 
    \end{definition}

\begin{definition}\label{def:norm2}
   Se define la \emph{norma 2} o \emph{norma euclidiana}, la cual notaremos por  $\|\;\|_2$, a
    \[
        \|\mathbf{x}\|_2 = \sqrt{ \sum_{i=1}^n x_i^2 }.
    \]
    donde $\mathbf{x}=(x_1,\ldots,x_n).$
\end{definition}

\begin{tarea} Probar que  $\| \; \|_1$ y  $\| \; \|_2$ son efectivamente normas, es decir, que cumplen con  lo pedido en (\ref{def:norma-general}). 
\end{tarea}

\begin{definition}
    Sean $\mathbf{x},\;\mathbf{y}\in\Rn{n}$. Se define la \emph{distancia} entre $\mathbf{x}\text{ e }\mathbf{y}$ como la funci\'on d$:\Rn{n}\times\Rn{n}\to\R$ tal que
    \[
        \text{d}(\mathbf{x},\mathbf{y})=\|\mathbf{x}-\mathbf{y}\|,
    \]
    donde $\|\cdot\|$ es  cualquier norma de $\Rn{n}$.
\end{definition}

\begin{definition} \textbf{Bola de radio $\delta$ y centro $x_o$.} \label{def:bola}
    Sean $x_o \in \Rn{n},\;\delta > 0$ y $\text{d}$ una distancia en $\Rn{n}$. La \emph{bola de radio $\delta$ y centro $x_o$}, que denotamos con $B_{\delta}(x_o)$, es el conjunto
    \[
     B_{\delta}(x_o) = \left\lbrace x \in \Rn{n} : \text{d}(x,x_o) < \delta \right\rbrace.
    \]
    El conjunto
    \[
     B^*_{\delta}(x_o) = B_{\delta}(x_o) \smallsetminus \{ x_o \}
    \]
 se llama \emph{bola reducida de radio $\delta$ y centro $x_o$}.
   
   \end{definition}

   \begin{example}
    Para ilustrar lo mencionado, podemos tomar el caso de dos bolas en $\Rn{2}$ centradas en el origen, de ``radio'' $\delta$ y con distintas nociones de distancia. La primera con distancia inducida por la norma 2 y la segunda con distancia inducida por la norma 1. Se deja al lector deducir anal\'iticamente los gr\'aficos mostrados a continuaci\'on.

   \begin{definition} \textbf{Punto interior, interior de un conjunto.} \label{def:interior}
    Sean $A  \subseteq  \Rn{n} \text{ y } \vect{x}_0 \in A$. Decimos que $\vect{x}_0$ es \emph{punto interior} de $A$ si
    \[
     \exists\:\delta > 0: B_{\delta}(\vect{x}_0)  \subseteq  A.
    \]
    El \emph{interior} de $A$, que denotamos con $A^{\circ}$, es el conjunto de todos los puntos interiores de $A$, es decir
    \[
    A^{\circ}= \left\lbrace\vect{x}\in A :\vect{x}\text{ es punto interior de } A \right\rbrace.
    \]
   \end{definition}
   
   
   \begin{definition} \textbf{Conjunto abierto.} \label{def:abierto}
       Sea $A  \subseteq  \Rn{n}$. Decimos que $A$ es \emph{abierto} si cumple que
   \[
        \forall \vect{x}_0 \in \exists \delta > 0 : B_{\delta}(\vect{x}_0)  \subseteq  A.  
%\footnote{\text{\textit{i.e.}: $A$  es abierto si todos sus puntos son interiores.}}
   \]
   \end{definition}
   
   \begin{propertie}  \label{prop:abierto_int}    Sea  $A \subseteq \Rn{n}$, entonces:
   \begin{enumerate}
    \item \( \interior{A} \subseteq A \).
    \item \(A\) es abierto \(\iff A=\interior{A}\).
    \end{enumerate}
    \end{propertie}
   
   \begin{definition} \textbf{Conjunto cerrado.} \label{def:cerrado}
    Sea $A  \subseteq  \Rn{n}$. Decimos que $A$ es \emph{cerrado} si su complemento $A^c = \Rn{n} \smallsetminus A$ es abierto.
   \end{definition}
   
   \begin{definition} \textbf{Frontera de un conjunto.} \label{def:frontera}
    Sea $A  \subseteq  \Rn{n}$. La \emph{frontera} de $A$, que denotamos con $\partial A$ o $\frontera(A)$, es el conjunto
    \[
     \partial A = \left\lbrace \vect{x}_0 \in \Rn{n} \;|\; \forall \delta > 0, \;
     B_{\delta}(\vect{x}_0) \cap A \ne \varnothing\;\land\;
     B_{\delta}(\vect{x}_0) \cap A^c \ne \varnothing \right\rbrace.
    \]
   \end{definition}
   
   \begin{propertie} \label{prop:cerrado_front}
    Un conjunto $A  \subseteq  \Rn{n}$ es cerrado si y solo si $\partial A  \subseteq  A$.
   \end{propertie}
   
   \begin{definition} \textbf{Punto de acumulaci\'on.} \label{def:pto_acum}
    Sean $A  \subseteq  \Rn{n} \text{ y } \vect{x}_0 \in \Rn{n}$. Decimos que $\vect{x}_0$ es \emph{punto de acumulaci\'on} de $A$ si
    \[
     \forall \delta > 0 \left( B_{\delta}(\vect{x}_0) - \{ \vect{x}_0 \} \right) \cap A \ne \varnothing. 
    \]
    Denotamos el conjunto de todos los puntos de acumulaci\'on de $A$ con $A'$:
    \[
     A' = \{\vect{x}\in \Rn{n} :\vect{x}\text{ es punto de acumulaci\'on de } A \}.
    \]
   
   \end{definition}
   
   \begin{definition} \textbf{Entorno de un punto.} \label{def:entorno}
    Sean $N  \subseteq   \Rn{n}$ y $\vect{x}_0 \in \Rn{n}$. Decimos que $N$ es \emph{entorno} de $\vect{x}_0$ si
    \[
     \exists \delta > 0 \;|\; B_{\delta}(\vect{x}_0)  \subseteq  N.
    \]
   \end{definition}
   
   \begin{definition} \textbf{Punto aislado.} \label{def:pto_ais}
    Sean $A  \subseteq   \Rn{n}$ y $\vect{x}_0 \in A$. Decimos que $\vect{x}_0$ es \emph{punto aislado} de $A$ si $\vect{x}_0$ no es punto de acumulaci\'on de $A$.
   \end{definition}
   
   \begin{definition} \textbf{Conjunto acotado.} \label{def:conj_acotado}
    Sea $A  \subseteq  \Rn{n}$. Decimos que $A$ es un \emph{conjunto acotado} si existe $\delta > 0$ tal que la bola de radio $\delta$  centrada en el origen contiene al conjunto $A$, es decir, si $\exists \delta > 0 : A  \subseteq  B_{\delta}(\mathbf{0})$.\\
 %   Equivalentemente, $A$ es un \emph{conjunto acotado} si $\exists \delta > 0, \vect{x}_0 \in \Rn{n} : A  \subseteq  B_{\delta}(\vect{x}_0)$.
   \end{definition}

\begin{definition} \textbf{Conjunto  conexo.} Sea   $A \subseteq  \mathbb{R}^{n}$. Decimos que $A$  es un conjunto \emph{conexo}  si   no existen $U,V  \subseteq  \mathbb{R}^{n} $  conjuntos abiertos, disjuntos, no vac\'ios   tales que  \(A \cap U \neq \emptyset\), \(A \cap V \neq \emptyset\) y \(A = U \cup V\).
\end{definition}
