\begin{obs} Si en la \autoref{def:grafm1} cambiamos  $A \subseteq \mathbb{R}$ por $A \subseteq \mathbb{R}^{n}$  obtenemos de manera an\'aloga la siguiente definici\'on.    
\end{obs}



\begin{definition}  \label{def:grafica}
 
 Sea \( f: A  \subseteq  \mathbb{R}^n \rightarrow \mathbb{R}\),  se define la \textit{gráfica de $f$}, y se nota $\text{Graf}(f)$,  a
 \[
\text{Graf}(f)=\{(x_1,x_2,...,x_n,x_{n+1})\in \mathbb{R}^{n+1} \;|\; (x_1,x_2,...,x_n) \in A \land x_{n+1}= f(x_1,x_2,...,x_n) \}
 \]

\end{definition}
\begin{obs}  Como \( f: A  \subseteq  \mathbb{R}^n \rightarrow \mathbb{R}\)  entonces $\text{Graf}(f)  \subseteq  \mathbb{R}^{n+1},$  con el objetivo de poder dibujar,  nos detendremos especialmente en el caso de gr\'aficas de  campos  $f: A\subseteq \mathbb{R}^2 \rightarrow \mathbb{R},$ cuya definici\'on se desprende automaticamente  de la definici\'on  (\ref{def:grafica}),  es decir, 
 \[
\text{Graf}(f)=\{(x,y,z)\in \mathbb{R}^3 \;|\;(x,y)\subseteq A \land z=f(x,y) \}.
 \]
\end{obs}
