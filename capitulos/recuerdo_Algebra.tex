\begin{definition}
  Se define a $\mathbb{R}^n$  como el conjunto formado por todas las $n$-tuplas de n\'umeros reales ordenados, \textit{i.e.}
    \begin{equation*}
        \mathbb{R}^n = \left\{ \vect{v}\;\rvert\;\vect{v}=(x_1,x_2,x_3,...,x_n), \; x_i \in \mathbb{R} \;\forall \;   1\leq  i \leq n  \right\}.
    \end{equation*}
\end{definition}

\begin{definition}{\textbf{Transformación Lineal.}}
  Sean  $\mathbb{V}$ y $\mathbb{W}$ sub-espacios vectoriales de $\mathbb{R}^n$ y  $\mathbb{R}^m$ respectivamente.  Se  define a  una \textit{transformación lineal}  como a  una función 
    \begin{equation*}
       \campoVec{T}: \mathbb{V} \rightarrow  \mathbb{W}
    \end{equation*}
    que cumple con la condición de linealidad, es decir,  para todos $\vect{v}_1,\vect{v}_2 \in \mathbb{V}$ y para todos $\alpha,\beta \in \mathbb{R}$. 
    \begin{equation}\label{tl}
        \campoVec{T}(\alpha\vect{v}_1 + \beta\vect{v}_2)=\alpha\campoVec{T}(\vect{v}_1) + \beta\campoVec{T}(\vect{v}_2),
    \end{equation}
 \label{def:TL}
\end{definition}

\begin{definition}{\textbf{Producto Escalar.}}
    Se define el \textit{producto escalar}, o \textit{producto interno canónico} en $\mathbb{R}^n$ para dos vectores
    $\vect{u}=(u_1, u_2,\ldots, u_n )$ y $\vect{v}=(v_1,v_2,\ldots,v_n)$ como
    \begin{equation*}
      \vect{u} \cdot \vect{v} = u_1v_1 + u_2v_2 + \ldots + u_nv_n = \sum_{i = 1}^{n} u_iv_i . 
    \end{equation*}
\end{definition}

\begin{definition} {\textbf{Recta en el espacio.}}
  Sean $\vect{v}\in\mathbb{R}^n$   no nulo y  $\vect{x_0}\in\mathbb{R}^n$.  El conjunto $L \subset \mathbb{R}^n $ dado por
 \begin{equation*}
      L = \{   \vect{x} \in\mathbb{R}^n  \;\rvert\;    \vect{x}=  \lambda  \vect{v} + \vect{x_0},  \:  \mbox{  con }  \lambda\in\mathbb{R} \},
    \end{equation*}  es una \textit{recta} en $ \mathbb{R}^n$. La ecuaci\'on $$ \vect{x}=  \lambda  \vect{v} + \vect{x_0}$$  es la  \emph{expresi\'on vectorial} de la misma.  
    Adem\'as, llamaremos al vector $\vect{x_0}$ como \emph{punto de paso} de la recta y al vector $\vect{v}$ como \emph{vector director} de la recta.
 \end{definition}

\begin{definition}{\textbf{Plano en el espacio.}}\label{plano}
  Sean    $\vect{n}\in\mathbb{R}^3$   no nulo  y  $\vect{x_0}\in\mathbb{R}^3,$  el conjunto $\Pi \subset \mathbb{R}^3 $  dado por 
    \begin{equation*}
      \Pi  = \{   \vect{x} \in\mathbb{R}^3  \;|\;    \vect{n}\cdot(\vect{x} - \vect{x_0}) =0   \}.
    \end{equation*} es un \textit{plano} en $\mathbb{R}^3$. La ecuaci\'on   $$ \vect{n}\cdot(\vect{x} - \vect{x_0}) =0$$  es la  \emph{expresi\'on canónica} del mismo.  Adem\'as, llamaremos al vector $\vect{n}$   como \emph{vector normal} del plano  y al vector  $\vect{x_0}$ \emph{punto de paso} del plano.   

    A su vez,  se puede expresar un plano por:
    \begin{equation*}
      \Pi  = \{   \vect{x} \in\mathbb{R}^3  \;|\;    \vect{x}=\vect{x_0}+\alpha \vect{u} + \beta \vect{u},  \:  \mbox{  con }  \alpha,\beta\in\mathbb{R}    \}.
    \end{equation*} La ecuación
    \begin{equation*}
       \vect{x}=\vect{x_0}+\alpha \vect{u} + \beta \vect{v}
    \end{equation*}
    es  la \emph{forma vectorial} del plano,  donde $\vect{u}$ y $\vect{v}$ son dos \emph{vectores paralelos al plano} tales que  $\vect{v}\nparallel\vect{u}$.
  \label{def:planoR3}
 \end{definition}

 \begin{obs}
    Notar que se cumple $\vect{v}\times\vect{u}=\lambda\vect{n}$, con $\lambda\in\mathbb{R}\smallsetminus\{0\}$.
 \end{obs}

\begin{definition}{\textbf{Circunferencia y Círculo en el plano.}} 
  Sean  $(x_0,y_0) \in \mathbb{R}^2$ y $r \in \mathbb{R}_{ > 0}. $
  El conjunto $C \subset \mathbb{R}^2$ dado por
    \begin{equation}\label{circ}
    C = \{  (x,y)  \in \mathbb{R}^2 \:|\:   (x-x_0)^2+(y-y_0)^2=r^2 \}
    \end{equation}  es una \textit{circunferencia en el plano}. Adem\'as, al vector $(x_0,y_0)$  se lo llama centro y al n\'umero $r$  se lo llama radio de la misma.
   
  El conjunto $D \subset \mathbb{R}^2$ dado por
    \begin{equation*}
    D = \{  (x,y)  \in \mathbb{R}^2 \:|\:   (x-x_0)^2+(y-y_0)^2 \leq r^2 \}
    \end{equation*}  es un \textit{c\'irculo en el plano}. Adem\'as, al vector $(x_0,y_0)$  se lo llama centro y al n\'umero $r$  se lo llama radio del mismo.  
   \end{definition}

\begin{definition}\textbf{Esfera y Superficie esférica.}
    El conjunto $S \subset \mathbb{R}^3$ dado por 
    \begin{equation*}
      S = \{  (x,y,z)  \in \mathbb{R}^3 \:|\:   (x-x_0)^2+(y-y_0)^2+(z-z_0)^2=r^2 \}.
    \end{equation*}
    es una Superficie Esférica, al punto $(x_0,y_0,z_0)$ se lo llama centro  y al n\'umero $r$ se lo llama radio,  de la misma.
  
  
    El conjunto  $E\subset\mathbb{R}^3$  dado por 
    \begin{equation*}
      E = \{  (x,y,z)  \in \mathbb{R}^3 \:|\:   (x-x_0)^2+(y-y_0)^2+(z-z_0)^2\leq r^2 \}.
    \end{equation*}  es una Esfera, y analogamente,   al punto $(x_0,y_0,z_0)$ se lo llama centro  y al n\'umero $r$ se lo llama radio  de la misma.
  \end{definition}

\begin{obs} Podemos pensar a la \textbf{Superficie esférica} como la  \textbf{ ``c\'ascara'' de la Esfera} encerrada por ella. 
\end{obs}
